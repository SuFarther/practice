[索引数组]
   下标就是数字开始的
   $arr = ['a','b','c',1,2,3];


[关联数组]
   $arr = [
       'a' => 'b',
       'c' => 'd',
       'e' => 'f'
   ]

[二维数组] 关联和索引混合的
   $arr = [
      'a',
      'b',
      'c',
      'd' => [
          'a' => 'b',
          'e',
          'f',
          'g'
      ],
      'aa',
   ];
